int ctrlDigit(string text, int& error) {
    int m[3] = { 7, 3, 1 };
    int n = 0;

    for (int i = 0; text[i] != '\0'; i++) {
        if (isdigit(text[i])) {
            n += (text[i] - '0') * m[i % 3];
        } else if (isalpha(text[i])) {
            n += (toupper(text[i]) - 'A') * m[i % 3];
        } else {
            error = i;
            return -1;
        }
    }

    return n % 10;
}
